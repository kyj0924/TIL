  <div id="gal">
    <div class="p">
      <span class="p-num">
        <span></span> / <span></span>
      </span>
      <button class="prev">&lt;</button>
      <button class="next">&gt;</button>
    </div>
   	<div class="con">
		<div><img src="images/Road_Repair.jpg"></div>
		<div><img src="images/Road_Sign.jpg"></div>
		<div><img src="images/Street_Lamp.jpg"></div>	
	</div>
</div>

<script>
	// 이미지 개수
	var total = $('.con div').length; // 3
	var current=1;
	
	// class p-num에 있는 첫번째 span에 적용
	$('.p-num span:first').text(current); // 1
	$('.p-num span:last').text(total); // 3
	
	// 다음 버튼 누를 때
	$('.next').click(function(){
		current++;
	
		if(current > total){
			current=1;
		}
		
		// insertAfter는 지정한 요소 뒤에 새로운 요소를 삽입한다.
		// class con에 있는 마지막 div 뒤에 첫번째 div를 추가한다는 의미
		$('.con > div:first').insertAfter('.con > div:last');
		// class p-num에 있는 첫번째 span의 text로 current를 띄운다.
		$('.p-num span:first').text(current);
		$('.con > div:first').css('display', '')
		$('.con > div:first').siblings().css('display', 'none');
	});
	
	// 이전 버튼 누를 때
	$('.prev').click(function(){
		current--;
		
		if(current < 1){
			current=total;
		}
		
		// insertAfter는 지정한 요소 뒤에 새로운 요소를 삽입한다.
		// class con에 있는 첫번째 div 앞에 마지막 div를 추가한다는 의미
		$('.con > div:last').insertBefore('.con > div:first');
		// class p-num에 있는 첫번째 span의 text로 current를 띄운다.
		$('.p-num span:first').text(current);
		$('.con > div:first').css('display', '')
		$('.con > div:first').siblings().css('display', 'none');
	});
	
	// 웹 문서 로딩될 때 첫번째 이미지만 표시
	$('document').ready(function(){
		$('.con > div:first').css('display', '')
		$('.con > div:first').siblings().css('display', 'none');
	});
	
	
</script>

	$('.btn-toggle').click(function(){
		// toggleClass는 함수를 넣었다 뺐다 하는 것
		$('body').toggleClass('dark-mode');
		$(this).css({'border':'1px solid #ccc'});

## ajax(Asynchronous javascript and xml)

동적 웹 페이지 개발하는 기법
웹페이지 전체를 로드하지 않고, 웹 페이지의 일부분만을 갱신할 수 있다.

서버와 다양한 형태의 데이터를 주고 받을 수 있는데, json, xml, html, txt 파일 등이 가능하다.
